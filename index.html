üåê JavaScript (Fetch API)

// Fetch all content
fetch('/api/content.json')
  .then(response => response.json())
  .then(data => {
    console.log('All content:', data);
  });

// Find specific content by slug
fetch('/api/content.json')
  .then(response => response.json())
  .then(data => {
    const post = data.find(item => item.slug === 'your-slug-here');
    console.log('Specific post:', post);
  });

üü¢ Node.js

const https = require('https');

// Fetch all content
https.get('https://yourdomain.com/api/content.json', (res) => {
  let data = '';
  res.on('data', chunk => data += chunk);
  res.on('end', () => {
    const content = JSON.parse(data);
    console.log('All content:', content);
  });
});

// Using axios (npm install axios)
const axios = require('axios');

async function getContent() {
  try {
    const response = await axios.get('https://yourdomain.com/api/content.json');
    return response.data;
  } catch (error) {
    console.error('Error fetching content:', error);
  }
}
